## linux 정리

> 다음 문서는 생활코딩의 linux 강의를 수강하며 배운 linux 명령어들을 정리한 것이다.

```shell
$ pwd # 현재 디렉토리의 위치를 알려준다
$ ls # 현재 디렉토리 안의 파일, 폴더 확인
$ ls -l # 하위 디렉토리 자세히 조회 # l은 list!
$ ls -a # 숨김파일(.로 시작)까지 모두 조회
$ ls -al # 위의 두 명령어 병합
$ ls -S # 파일 크기를 기준으로 정렬해 보여줌
$ ls -rtl # 파일이 업데이트된 기준대로 정렬해 보여줌
$ mkdir [폴더 이름] # 디렉토리 생성
$ mkdir -p [생성하고 싶은 폴더 경로] # 경로에 새로운 부모 디렉토리가 없을 경우 알아서 생성
$ touch empty_file.txt # 비어있는 파일 생성
$ cd .. # 상위 디렉토리로 이동
$ cd / # 최상위 디렉토리로 이동
$ clear # 커맨드 창을 청소 # 첫 줄 외의 모든 커맨드를 지워버림
$ rm [삭제할 파일.파일 확장자] # 파일 삭제
$ rm -r [삭제할 폴더] # 폴더 삭제 # -r 옵션을 붙이지 않으면 폴더 내의 모든 디렉토리 & 파일 삭제 안됨
$ [명령어] --help # 해당 명령어에 대한 레퍼런스 조회(레퍼런스가 화면에 표시됨) # ex> rm --help
$ man [명령어] # 해당 명령어에 대한 레퍼런스 조회(레퍼런스 페이지로 넘어감) # /[검색어] : 검색어가 포함된 부분만 검색 가능
# n을 눌러 검색어가 포함된 다음 줄로 넘어갈 수 있다 # q : 원래 커맨드 창으로 돌아온다
$ tail [파일명] # 해당 파일의 끝 10줄만 보여줌
$ tail -n 20 [파일명] # 해당 파일의 끝 20줄까지만 보여줌
$ tail -f [파일명] # 해당 파일의 끝 10줄을 보여주고, 지속적으로 추가되는 내용까지 실시간으로 표시해줌
$ more [파일명] # 해당 파일 전체 내용을 보여줌 # 아래로 이동은 space bar # /로 특정 문자열 검색 가능
$ grep -r [찾을 문자열] [파일명] # 해당 파일에서 문자열이 위치하는 부분을 출력
$ ps -ef | grep [파일 확장자] # ps -ef | grep py
$ cat [파일명] # 특정 파일을 열어봄 # 텍스트 파일은 그대로 출력, 바이너리 파일은 출력해도 알아볼 수 없음
$ cat -b [파일명] # 각 행에 번호를 붙여 출력
$ cat -n [파일명] # 빈 행에도 번호를 붙여 출력함
$ less [파일명] # more과 유사
$ head [파일명] # 특정 파일을 가장 앞 10줄까지만 출력
$ head -n 20 [파일명] # 특정 파일의 가장 앞 20줄까지 출력
$ mv "현재 파일 이름" "새로운 파일 이름" # 파일 이름 변경 ex. mv old_file.txt new_file.txt
$ ps -e # 메모리 상에 올라와 있는 전체 프로세스 현황 출력 # pid 확인 가능
$ ps -f # 각각의 명령어에 대한 프로세스 id, 부모 프로세스 id 포함한 전체 리스트 표시
$ ps -ef # 현재 실행 중인 전체 프로세스, 프로세스 실행 파일 루트까지 전부 출력 # pid 확인 가능
$ kill -9 [종료할 프로세스 pid 번호] # 특정 프로세스 강제 종료
$ du [폴더 이름] # 해당 디렉토리와 하위 디렉토리까지의 디스크 용량 출력(kbyte 단위)
$ du -s [폴더 이름] # 해당 폴더 하나의 디스크 용량만을 출력
$ du -sh [폴더 이름] # 용량이 읽기 편한 단위로 출력
$ du -sh [폴더 이름]/* # 해당 디렉토리와 하위 디렉토리까지의 디스크 용량을 읽기 쉽게 출력
$ ls -d abc* # abc로 시작하는 파일 전부 출력
$ ls -d *abc* # 파일명에 abc를 포함하는 파일 전부 출력
$ ls -d *abc # 파일명이 abc로 끝나는 파일 전부 출력
$ df # 남은 메모리 용량 확인
$ df -h # 남은 메모리 용량을 깔끔하게 정리해서 보여줌
$ df . # 현재 경로의 디스크 용량만 출력
$ ll # 현재 경로의 파일당 용량을 정리해서 보여줌
$ du -h # 현재 경로 파일 b 단위로 보여줌(읽기 편하게)
$ du -b # 현재 경로 실제 파일 용량 보여줌
$ sudo mv [현재 파일 경로] [이동할 파일 경로] # 파일 위치 이동
$ ps -ef ? grep python # python 프로세스 정보 확인
$ ls -Art | tail -n [개수] # 가장 최근에 수정된 파일을 n개 보여준다
```
